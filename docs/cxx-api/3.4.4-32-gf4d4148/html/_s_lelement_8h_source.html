141 virtual const pair<string, string> generateJSON(
142 vector<const SLelement<E>*> nodes) const {
143 // map the nodes to a sequence of ids, 0...N-1
144 // then get the JSON string for nodes placeholder
145 // nullptr prevents insertion of other nullptrs
146 unordered_map<const SLelement*, int> node_map { {nullptr, -1} };
147
148 string nodes_JSON, links_JSON;
149
150 int i = 0; // get the JSON string for nodes
151 for (const auto* e : nodes) {
152 if (node_map.emplace(e, i).second) {
153 // successful emplacement
154 i++;
155 nodes_JSON += e->getElementRepresentation() + COMMA;
156 }
157 }
158 //Remove trailing comma and nullptr entry
159 node_map.erase(nullptr);
160 if (nodes_JSON.size()) {
161 nodes_JSON = nodes_JSON.erase(nodes_JSON.size() - 1);
162 }
163 // for each pair<SLelement*,int> in map
164 for (unsigned int k = 0; k < nodes.size(); k++) {
165 if (nodes[k]->next != nullptr) { // link exists
166 links_JSON += this->getLinkRepresentation(nodes[k]->links.at(nodes[k]->next),
167 to_string(node_map[nodes[k]]),
168 to_string(node_map[nodes[k]->next]) ) + COMMA;
169 }
170 }
171
172 //Remove trailing comma
173 if (links_JSON.size()) {
174 links_JSON = links_JSON.erase(links_JSON.size() - 1);
175 }
