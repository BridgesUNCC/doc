660 QUOTE + "map_overlay" + QUOTE + COLON +
661 ((map_overlay) ? "true" : "false") + COMMA +
662 QUOTE + "map" + QUOTE + COLON + QUOTE + map + QUOTE + COMMA;
663
664 json_header += QUOTE + "element_label_flag" + QUOTE + COLON + ((element_labelFlag) ? "true" : "false") + COMMA +
665 QUOTE + "link_label_flag" + QUOTE + COLON + ((link_labelFlag) ? "true" : "false") + COMMA +
666 QUOTE + "coord_system_type" + QUOTE + COLON + JSONencode(getCoordSystemType()) +
667 COMMA;
668
669 if (wc_window.size() == 4) { // world coord window has been specified
670 json_header += QUOTE + string("window") + QUOTE + COLON + OPEN_BOX;
671 json_header += std::to_string(wc_window[0]) + COMMA +
672 std::to_string(wc_window[1]) + COMMA +
673 std::to_string(wc_window[2]) + COMMA + std::to_string(wc_window[3]);
674 json_header += CLOSE_BOX + COMMA;
