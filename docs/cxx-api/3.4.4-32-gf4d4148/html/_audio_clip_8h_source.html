306 void parseWaveFile (const string & wave_file) {
307 // Read and parse an audio file in WAVE format
308
309 // open file
310 ifstream infile;
311 infile.open (wave_file.c_str(), ios::binary | ios::in);
312 if (infile.fail()) {
313 throw "Could not open " + wave_file;
314 }
315
316 // read the header data of the input WAVE file
317 WaveHeader wave_header = readWaveHeader(infile);
318
319 long size_of_each_sample = (wave_header.channels *
320 wave_header.bits_per_sample) / 8;
321
322 // read the audio data
323 if (this->sampleCount > 1000000000) {
324 throw "sampleCount must be less than 1 million";
325 }
326
327 // create storage for the audio data
328 //this->channels.resize(this->numChannels);
329 for (int i = 0; i < numChannels; i++) {
330 this->channels.push_back( AudioChannel(this->sampleCount));
331 }
332
333 // read sample data by chunks, if PCM
334 if (wave_header.format_type == 1) { // Only PCM handled
335 long i = 0;
336 char data_buffer[size_of_each_sample];
337 int size_is_correct = true;
338
339 // make sure that the bytes-per-sample is completely divisible
340 // by num.of channels
341 long bytes_in_each_channel = (size_of_each_sample / wave_header.channels);
342 if ((bytes_in_each_channel * wave_header.channels) != size_of_each_sample) {
343 cout << "Error: Incorrect chunk size.. " << bytes_in_each_channel
344 << ", " << wave_header.channels << ", " << size_of_each_sample << "\n";
345 size_is_correct = false;
346 }
347
348 if (size_is_correct) {
349 // the valid amplitude range for values based on the bits per sample
350 long low_limit = 0l;
351 long high_limit = 0l;
352
353 switch (wave_header.bits_per_sample) {
354 case 8:
355 low_limit = -128;
356 high_limit = 127;
357 break;
358 case 16:
359 low_limit = -32768;
360 high_limit = 32767;
361 break;
362 case 24:
363 low_limit = -16777216;
364 high_limit = 16777215;
365 break;
366 case 32:
367 low_limit = -2147483648;
368 high_limit = 2147483647;
369 break;
370 }
371 for (int sample = 0; sample < this->sampleCount;
372 sample++) {
373 int amplitude;
374 if (!infile.fail()) {
375 for (int ch = 0; ch < wave_header.channels;
376 ch++ ) {
377 // read signal amplitude
378 infile.read(data_buffer, bytes_in_each_channel);
379 // convert data from big endian to little
380 // endian based on bytes in each channel sample
381 switch (bytes_in_each_channel) {
382 case 1:
383 amplitude = data_buffer[0] & 0x00ff;
384 amplitude -= 128; //in wave, 8-bit are unsigned, so shifting to signed
385 break;
386 case 2:
387 amplitude =
388 (data_buffer[0] & 0x00ff) |
389 (data_buffer[1] << 8);
390 break;
391 case 3:
392 amplitude =
393 (data_buffer[0] & 0x00ff) |
394 ((data_buffer[1] & 0x00ff) << 8) |
395 (data_buffer[2] << 16);
396 break;
397 case 4:
398 amplitude =
399 (data_buffer[0] & 0x00ff) |
400 ((data_buffer[1] & 0x00ff) << 8) |
401 ((data_buffer[2] & 0x00ff) << 16) |
402 (data_buffer[3] << 24);
403 break;
404 }
405 this->setSample(ch, sample, amplitude);
406 }
407 }
408 else {
409 cout << "Error reading file\n.";
410 break;
411 }
412 }
413 }
414 }
415 infile.close();
416 }

422 WaveHeader readWaveHeader(ifstream& infile) {
423
424 // read file header
425 WaveHeader wave_header;
426
427 infile.read ((char *)wave_header.riff, 4);
428
429 if (wave_header.riff[0] != 'R' ||
430 wave_header.riff[1] != 'I' ||
431 wave_header.riff[2] != 'F' ||
432 wave_header.riff[3] != 'F')
433 throw "malformed RIFF header";
434
435 unsigned char *buffer = new unsigned char[4];
436 infile.read ((char*) buffer, 4);
437
438 // convert little endian to big endian 4 byte int
439 wave_header.overall_size = buffer[0] | (buffer[1] << 8) |
440 (buffer[2] << 16) | (buffer[3] << 24);
441
442 if (debug)
443 std::cout << "overall size: " << wave_header.overall_size << std::endl;
444
445 infile.read ((char*) wave_header.wave, 4);
446
447 if (wave_header.wave[0] != 'W' ||
448 wave_header.wave[1] != 'A' ||
449 wave_header.wave[2] != 'V' ||
450 wave_header.wave[3] != 'E')
451 throw "format is not WAVE";
452
453 infile.read ((char*) wave_header.fmt_chunk_marker, 4);
454 if (wave_header.fmt_chunk_marker[0] != 'f' ||
455 wave_header.fmt_chunk_marker[1] != 'm' ||
456 wave_header.fmt_chunk_marker[2] != 't' ||
457 wave_header.fmt_chunk_marker[3] != ' ')
458 throw "malformed wave file";
459
460 infile.read ((char *) buffer, 4);
461 wave_header.length_of_fmt = buffer[0] | (buffer[1] << 8) |
462 (buffer[2] << 16) | (buffer[3] << 24);
463
464 if (debug)
465 std::cout << "length of format: " << wave_header.length_of_fmt << std::endl;
466
467 char *buffer2 = new char[2];
468 infile.read (buffer2, 2);
469 wave_header.format_type = buffer2[0] | (buffer2[1] << 8);
470
471 string format_name = "";
472 switch (wave_header.format_type) {
473 case 1 :
474 format_name = "PCM";
475 break;
476 case 6 :
477 format_name = "A-law";
478 break;
479 case 7 :
480 format_name = "Mu-law";
481 break;
482 default:
483 throw "unsupported format";
484
485 }
486
487 infile.read (buffer2, 2);
488 wave_header.channels = buffer2[0] | (buffer2[1] << 8);
489 this->numChannels = wave_header.channels;
490
491 if (debug)
492 std::cout << "numChannels: " << numChannels << std::endl;
493
494 infile.read ((char *) buffer, 4);
495 wave_header.sample_rate = buffer[0] | (buffer[1] << 8) |
496 (buffer[2] << 16) | (buffer[3] << 24);
497 this->sampleRate = wave_header.sample_rate;
498
499 if (debug)
500 std::cout << "sampleRate: " << sampleRate << std::endl;
501
502 infile.read ((char *) buffer, 4);
503 wave_header.byterate = buffer[0] | (buffer[1] << 8) |
504 (buffer[2] << 16) | (buffer[3] << 24);
505
506 if (debug)
507 std::cout << "byte rate: " << wave_header.byterate << std::endl;
508
509 infile.read (buffer2, 2);
510 wave_header.block_align = buffer2[0] | (buffer2[1] << 8);
511
512 infile.read (buffer2, 2);
513 wave_header.bits_per_sample = buffer2[0] | (buffer2[1] << 8);
514
515 this->sampleBits = wave_header.bits_per_sample;
516
517 if (debug)
518 std::cout << "sample Bits: " << sampleBits << std::endl;
519
520 if (wave_header.byterate !=
521 wave_header.sample_rate * wave_header.channels * wave_header.bits_per_sample / 8)
522 throw "malformed wave file";
523
524 //The next sub-chunk of the RIFF format should be the data subchunk.
525 //but in some case, there are meta data found first.
526 //skipping any sub-chunk that is not the data sub-chunk
527 bool data_chunk_found = false;
528 while (!data_chunk_found) {
529 infile.read ((char *)wave_header.data_chunk_header, 4);
530
531 infile.read ((char *) buffer, 4);
532 wave_header.data_size = buffer[0] | (buffer[1] << 8) |
533 (buffer[2] << 16) | (buffer[3] << 24);
534
535 if (wave_header.data_chunk_header[0] != 'd' ||
536 wave_header.data_chunk_header[1] != 'a' ||
537 wave_header.data_chunk_header[2] != 't' ||
538 wave_header.data_chunk_header[3] != 'a') {
539 //skip sub chunk
540 int padding = (wave_header.data_size % 2 ? 0 : 1);
541 infile.ignore(wave_header.data_size + padding);
542 }
543 else
544 data_chunk_found = true;
545 }
546
547 // calculate no.of samples
548 long num_samples = (8 * wave_header.data_size) /
549 (wave_header.channels * wave_header.bits_per_sample);
550 this->sampleCount = num_samples;
551
552 if (debug)
553 std::cout << "sample Count: " << this->sampleCount << std::endl;
554
555 long size_of_each_sample = (wave_header.channels *
556 wave_header.bits_per_sample) / 8;
557
558 delete[] buffer;
559 delete[] buffer2;
560
561 if (!infile)
562 throw "malformed RIFF header";
563
564 return wave_header;
565 }
